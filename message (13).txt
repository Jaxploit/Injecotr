local function UZRJMT_fake_script() -- ExecButton.LocalScript 
	local script = Instance.new('LocalScript', ExecButton)

	script.Parent.MouseButton1Click:Connect(function()
		local code = script.Parent.Parent.Editor.Text
		local success, errorMessage = pcall(function()
			loadstring(code)()
		end)
		if not success then
			warn("Error executing script: " .. errorMessage)
		end
	end)
	
end
coroutine.wrap(UZRJMT_fake_script)()

local function ZEYOOE_fake_script() -- Editor.LocalScript 
	local script = Instance.new('LocalScript', Editor)

	local lastKnownText = ""
	
	local function restoreLastText()
		script.Parent.Text = lastKnownText
		script.Parent.CursorPosition = #script.Parent.Text + 1
	end
	
	script.Parent.FocusLost:Connect(function()
		lastKnownText = script.Parent.Text
	end)
	
	script.Parent.Focused:Connect(function()
		if lastKnownText ~= "" then
			restoreLastText()
		end
	end)
end
coroutine.wrap(ZEYOOE_fake_script)()
local function LLJNN_fake_script() -- Frame.LocalScript 
	local script = Instance.new('LocalScript', Frame)

	local UserInputService = game:GetService("UserInputService")
	local dragging, dragInput, dragStart, startPos
	local frame = script.Parent
	
	local function update(input)
		local delta = input.Position - dragStart
		frame.Position = UDim2.new(startPos.X.Scale, startPos.X.Offset + delta.X, startPos.Y.Scale, startPos.Y.Offset + delta.Y)
	end
	
	frame.InputBegan:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseButton1 or input.UserInputType == Enum.UserInputType.Touch then
			dragging = true
			dragStart = input.Position
			startPos = frame.Position
	
			input.Changed:Connect(function()
				if input.UserInputState == Enum.UserInputState.End then
					dragging = false
				end
			end)
		end
	end)
	
	frame.InputChanged:Connect(function(input)
		if input.UserInputType == Enum.UserInputType.MouseMovement or input.UserInputType == Enum.UserInputType.Touch then
			dragInput = input
		end
	end)
	
	UserInputService.InputChanged:Connect(function(input)
		if input == dragInput and dragging then
			update(input)
		end
	end)
end
coroutine.wrap(LLJNN_fake_script)()
